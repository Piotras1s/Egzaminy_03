SELECT miasta.nazwa, wojewodztwa.nazwa FROM `miasta`
JOIN wojewodztwa
WHERE miasta.nazwa LIKE "Lu%"
AND miasta.id_wojewodztwa=wojewodztwa.id
ORDER BY miasta.nazwa ASC;
