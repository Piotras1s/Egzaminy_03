SELECT osoby.id, osoby.nazwisko, hobby.nazwa FROM osoby 
JOIN hobby ON hobby.id = osoby.id 
WHERE osoby.rok_urodzenia > 2000;

WHERE imie = 'monika' LIMIT 5;

wpis text,
uzytkownik_id int NOT null);
